#!/bin/sh

#AUTHOR : Tim Pattison
#SCRIPT : Activity12 - Compare Files


while read line
do
	for word in $line
	do
		words1+=($word)
	done
done <$1

while read line
do
	for word in $line
	do
		words2+=($word)
	done
done <$2

commonwords=()
for w in "${words1[@]}"
do
	for w2 in "${words2[@]}"
	do
		if [[ $w = $w2 ]]
		then
			commonwords+=($w)
		fi
	done
done

if [[ "${#commonwords[@]}" = 0 ]]
then 
	echo "There are no common elements between the two files."
else
	echo "The files share common words."
	echo "The common words are: ${commonwords[@]}"
fi
