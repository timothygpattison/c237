#!/bin/sh

#AUTHOR : Tim Pattison
#SCRIPT : Activity10 - Create Random Number File

if test -f "$1"
then
    	echo "Error. There is already a file with the name $1."
else

	touch $1
	re='^[0-9]+$'
	if [[ ! $2 =~ $re ]]
	then
        	echo "Error. Second parameter must be an integer."
	else
        	for i in $(seq 1 $2)
        	do
                	echo $RANDOM >> $1
        	done
	fi
fi
