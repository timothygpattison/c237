read -p "Please enter a number to compare " number1
while [[ ! $number1 =~ ^[0-9]+$ ]]
do
	read -p "Invalid. Please enter a number " number1
done
read -p "Please enter another number to compare " number2

while [[ ! $number2 =~ ^[0-9]+$ ]]
do
	read -p "Invalid. Please enter a number " number2
done

if [[ $number1 -gt $number2 ]]
then
	echo "The first number is larger"
elif [[ $number1 -lt $number2 ]]
then
	echo "The second number is larger"
else
	echo "The numbers are equal"
fi
